Methods of Array

  • 1.contact(), array1.contact(array2, array3)

    1. 链接多个数组
    2. 不会改变现有数组，只返回被连接数组的副本
    let [arr0, arr1, arr2] = [[1,2], [3,4], [5,6]]
    let arr3 = arr0.concat(arr1, arr2)
    console.log(arr0) \\[1,2]
    console.log(arr1) \\[3,4]
    console.log(arr2) \\[5,6]
    console.log(arr3) \\[1,2,3,4,5,6]

  • 2.copyWithin, array1.copyWithin(target, start, end)

    1. 复制数组指定位置的内容到数组的另一个指定位置中，会改变现有数组
    2. target:必需，复制到指定目标索引位置
    3. start:可选，复制的起始位置
    4. end:可选，复制的停止位置，负数为倒数,即：length + end
    let arr = [0,2,4,6,8,10]
    arr.copyWithin(3) \\ [0,2,4,0,2,4]
    arr.copyWithin(3,1) \\ [0,2,4,2,4,6]
    arr.copyWithin(3,1,3) \\ [0,2,4,2,4,10]
    arr.copyWithin(3,1,-1) \\ [0,2,4,2,4,6]

  • 3.entries(), array.entries()

    1. 返回键名和键值的Iterator集合
    2. 可以替换 for(let i = 0; i < array.length; i ++) 的写法
    let arr = ['a', 'b', 'c']
    arr.entries()
    [0,'a']
    [1,'b']
    [2,'c']
    for(let [index, val] of arr.entries()){
    console.log(`${index}:${val}`)
    }
    0:a
    1:b
    2:c

  • 4.every(), array.every(func)

    1. 检测所有元素是否符合条件，返回boolean
    2. 当遇到不符合条件的值就会中断下面的判断
    let arr = [12,14,18,20]
    let result1 = arr.every(i => i <= 20) \\ true
    let result2 = arr.every(i => i > 20) \\ false

  • 5.fill(), array.fill(value, start, end)

    1. 用一个固定值替换数组的元素
    2. value, 必填
    3. start, 非必填 开始填充的位置
    4. end, 非必填 结束填充的位置
    let arr = [12,14,18,20]
    arr.fill(20) \\ [20,20,20,20]
    arr.fill(20,0,1) \\ [20,14,18,20]

  • 6.find(), array.find(func(currentVal, index, arr),value)

    1. 返回符合条件的第一个值，遇到符合条件的值后不再执行判断
    2. 无匹配返回undefined
    3. currentVal，必填
    4. index，非必填
    5. arr，非必填
    6. value，非必填，如果传空，undefined会传给this
    let arr = [12,14,18,20]
    arr.find(currentVal => currentVal > 16) \\18

  • 7.findIndex(), array.findIndex(func(currentVal, index, arr),value)

    1. 返回符合条件的第一个值的下标，遇到符合条件的值后不再执行判断
    2. 无匹配返回-1
    3. currentVal，必填
    4. index，非必填
    5. arr，非必填
    6. value，非必填，如果传空，undefined会传给this
    let arr = [12,14,18,20]
    arr.findIndex(currentVal => currentVal > 16) \\2

  • 8.forEach(), array.forEach(func(currentVal, index, arr),value)

    1. 遍历所有元素
    2. currentVal，必填
    3. index，非必填
    4. arr，非必填
    5. value，非必填，如果传空，undefined会传给this
    let arr = [12,14,18,20]
    let result1 = arr.forEach((currentVal, index) => {
    console.log(`${index}:${currentVal}`)
    })
    0:12
    1:14
    2:18
    3:20

  • 9.from(), Array.from(object,mapFunc,value)

    1. 可将类似数组的对象和可遍历的对象转成数组
    2. object,必填
    3. mapFunc,非必填
    4. value,非必填
    let setObj = new Set(['a','b','c'])
    let arr = Array.from(setObj) \\['a','b','c']
    let arr = Array.from(setObj,i => `arr+${i}`) \\['arr+a','arr+b','arr+c']

  • 10.includes, array.includes(el, start)

    1. 判断数组是否包含指定的值，返回boolean
    2. start，非必填 开始检索的下标，负数为倒数，即len+start
    let arr = ['a','b','c']
    arr.includes('a') \\ true
    arr.includes('d') \\ false
    arr.includes('a',1) \\ false

  • 11.indexOf(), array.indexOf(el, start)

    1. 返回指定值第一个匹配到的小标，未匹配返回-1
    2. start，非必填 开始检索的下标
    let arr = ['a','b','c']
    arr.indexOf('a') \\ 0
    arr.indexOf('d') \\ -1

  • 12.isArray(), Array.isArray(obj)

    1. 判断对象是否为数组

  • 13.join(), array.join(separator)

    1. 将数组按separator转成字符串
    2. separator,可选 默认 ‘,’

  • 14.keys(), array.keys()

    1. 返回键名的Iterator集合

  • 15.lastIndexOf(), array.lastIndexOf(el,start)

    1. indexOf 的倒数

  • 16.map(), array.findIndex(func(currentVal, index, arr),value)

    1. 遍历所有元素,返回所有条件判断的数组集合
    2. currentVal，必填
    3. index，非必填
    4. arr，非必填
    5. value，非必填，如果传空，undefined会传给this
    let arr = [12,14,18,20]
    arr.map(i => i < 18) \\ [true,true,false,false]

  • 17.prop(), array.prop()

    1. 删除最后一个元素并返回删除的元素

  • 18.push(), array.push()

    1. 添加新元素并返回新的数组长度

  • 19.reduce(), array.reduce(func(total, currentVal, currentIndex, arr),initVal)

    1. 接受一个方法对数组元素进行累加，从左到右
    2. total，必填
    3. currentVal，必填
    4. currentIndex，非必填
    5. arr，非必填
    6. initVal，非必填，初始值
    let arr = [1,2,3,4]
    arr.reduce((total, currentVal) => total + currentVal) \\ 10
    arr.reduce((total, currentVal) => total + currentVal, 10) \\ 20

  • 20.reduceRight(),array.reduceRight(func(total, currentVal, currentIndex, arr),initVal)

    1. 同reduce，顺序从右到左

  • 21.reverse(), array.reverse()

    1. 颠倒数组中的顺序

  • 21.shift(), array.shift()

    1. 删除数组第一个元素并返回删除的值

  • 22.slice(), array.slice(start,end)

    1. 从已有数组中返回选定的元素集合
    2. 不会改变原始数组
    3. start, 可选
    4. end, 可选
    let arr = [1,2,3,4]
    arr.slice(1,3) \\ [2,3]

  • 23.some(),array.some(func(currentVal, index, arr),value)

    1. 对应every()，只要有一个元素符合条件就返回true

  • 24.sort(),array.sort(func)

    1. 排序顺序可以是字母或数字，并按升序或降序,默认排序顺序为按字母升序

  • 25.splice(), array.splice(index, howmany, items)

    1. 删除数组项并返回删除的元素集合
    2. index,必填 开始删除的下标
    3. howmany,必填 删除的个数
    4. items,选填 替换的元素

  • 26.toString(), array.toString()

    1. 等于 join()

  • 27.unshift(), array.unshift(...items)

    1. 向数组开头添加多个元素
